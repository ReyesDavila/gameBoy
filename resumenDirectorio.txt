     1	#include "../../../engine/game_bridge.h"
     2	
     3	void pelota_step(void) {
     4	
     5	        // --- MOVIMIENTO X ---
     6	
     7	    // Usamos XOR (^) para saber si solo uno está presionado (evita sumas y comparaciones)
     8	    uint8_t control = joystick_xpos(); 
     9	
    10	    if(control) {
    11	        // En lugar de: 1 - 2 * joy_l
    12	        // Usamos un simple if para asignar la dirección
    13	        if (joystick_xpos() == -1) pelota->dx = -1; else pelota->dx = 1;
    14	        
    15	        // En lugar de: pelota->vx += control * pelota->dx * pelota->ax
    16	        // Como sabemos que control es 1, sumamos o restamos la aceleración directamente
    17	        if (pelota->dx == 1) pelota->vx += pelota->ax; else pelota->vx -= pelota->ax;
    18	    } else {
    19	        // Fricción manual rápida (reemplaza sign() y multiplicaciones)
    20	        if (pelota->vx > 0) pelota->vx--;
    21	        else if (pelota->vx < 0) pelota->vx++;
    22	        
    23	        if (abs(pelota->vx) < 5) pelota->vx = 0;
    24	    }
    25	
    26	    // Límite de velocidad (reemplaza sign() * max)
    27	    if (pelota->vx > pelota->vx_max) pelota->vx = pelota->vx_max;
    28	    else if (pelota->vx < -pelota->vx_max) pelota->vx = -pelota->vx_max;
    29	
    30	    pelota->x2 += pelota->vx;
    31	    pelota->x = (pelota->x2 >> 5);
    32	
    33	    // --- MOVIMIENTO Y ---
    34	    uint8_t controly = joystick_ypos();
    35	
    36	    if(controly) {
    37	        if (joystick_ypos()==-1) pelota->dy = -1; else pelota->dy = 1;
    38	        
    39	        if (pelota->dy == 1) pelota->vy += pelota->ay; else pelota->vy -= pelota->ay;
    40	    } else {
    41	        if (pelota->vy > 0) pelota->vy--;
    42	        else if (pelota->vy < 0) pelota->vy++;
    43	        
    44	        if (abs(pelota->vy) < 5) pelota->vy = 0;
    45	    }
    46	
    47	    if (pelota->vy > pelota->vy_max) pelota->vy = pelota->vy_max;
    48	    else if (pelota->vy < -pelota->vy_max) pelota->vy = -pelota->vy_max;
    49	
    50	    pelota->y2 += pelota->vy;
    51	    pelota->y = (pelota->y2 >> 5);
    52	
    53	
    54	}

    16	    pool = storage;
    17	    max_entities = max_count;
    18	    entity_size = struct_size;
    19	    
    20	    for (uint8_t i = 0; i < max_entities; i++) {
    21	        InternalEntity* e = (InternalEntity*)((uint8_t*)pool + (i * entity_size));
    22	        e->active = 0;
    23	        move_sprite(i, 0, 0);
    24	    }
    25	}

    27	Entity* entity_create(int16_t x, int16_t y, uint8_t tile_id) {
    28	    for (uint8_t i = 0; i < max_entities; i++) {
    29	        InternalEntity* e = (InternalEntity*)((uint8_t*)pool + (i * entity_size));
    30	        if (!e->active) {
    31	            e->x = x; e->y = y;
    32	            e->tile_id = tile_id;
    33	            e->active = 1;
    34	            set_sprite_tile(i, tile_id);
    35	            return (Entity*)e;
    36	        }
    37	    }
    38	    return 0;
    39	}

    48	        uint8_t* tile_list;
    49	        uint8_t list_size;
    50	    } Renderable;
    51	
    52	    for (i = 0; i < max_entities; i++) {
    53	        Renderable* e = (Renderable*)((uint8_t*)pool + (i * entity_size));
    54	        
    55	        if (e->active) {

     4	uint16_t random(uint16_t x) {
     5	    return 0;
     6	    if (x == 0) return 0;
     7	    // rand() devuelve un valor entre 0 y RAND_MAX
     8	    // El operador % (módulo) limita el resultado al rango [0, x-1]
     9	    return (uint16_t)rand() % x;
    10

    13	// Raíz cuadrada optimizada sin usar punto flotante
    14	uint16_t sqrt_int(uint32_t x) {
    15	    uint32_t res = 0;
    16	    uint32_t bit = 1UL << 30; // El segundo bit más significativo de un uint32
    17	    while (bit > x) bit >>= 2;
    18	    while (bit != 0) {
    19	        if (x >= res + bit) { x -= res + bit; res = (res >> 1) + bit; }
    20	        else res >>= 1;
    21	        bit >>= 2;
    22	    }
    23	    return res;
    24	}
    25	
    26	// Eleva X a la potencia de N
    27	int32_t power(int16_t x, uint8_t n) {
    28	    int32_t res = 1;
    29	    for(uint8_t i = 0; i < n; i++) res *= x;
    30	    return res;
    31	}

    33	// Calcula la distancia usando Pitágoras (x^2 + y^2 = h^2)
    34	uint16_t point_distance(int16_t x1, int16_t y1, int16_t x2, int16_t y2) {
    35	    int32_t dx = (int32_t)(x2 - x1);
    36	    int32_t dy = (int32_t)(y2 - y1);
    37	    return sqrt_int((uint32_t)(dx * dx + dy * dy));
    38	}

    41	// ¡Esta es la magia de usar 256 pasos! Al restar y forzar a un entero con signo de 8 bits (int8_t), 
    42	// el hardware automáticamente encuentra el camino más corto hacia la izquierda o la derecha.
    43	int8_t angle_difference(uint8_t angle1, uint8_t angle2) {

    47	// Aproximación rápida del ángulo direccional (Devuelve 0-255)
    48	uint8_t point_direction(int16_t x1, int16_t y1, int16_t x2, int16_t y2) {
    49	    int16_t dx = x2 - x1;
    50	    int16_t dy = y2 - y1;
    51	    if (dx == 0 && dy == 0) return 0;
    52	    // Lógica básica de cuadrantes simplificada para Game Boy
    53	    if (abs(dx) > abs(dy)) {
    54	        if (dx > 0) return (dy > 0) ? 32 : 224; // Derecha (Aprox)
    55	        else return (dy > 0) ? 96 : 160;        // Izquierda (Aprox)
    56	    } else {
    57	        if (dy > 0) return 64;  // Abajo
    58	        else return 192;        // Arriba
    59	    }
    60	}

    32	void debug_watch(int16_t value) {
    33	    uint8_t buf[8]; 
    34	    uint8_t tmp[5]; 
    35	    int8_t p = 0;
    36	    uint8_t i = 0;

    38	    int16_t v = (value < 0) ? -value : value;
    39	
    40	    // Conversión de número a tiles

    42	        tmp[p++] = TILE_NUM_START;

    44	        while (v > 0 && p < 5) {
    45	            tmp[p++] = (v % 10) + TILE_NUM_START;
    46	            v /= 10;
    47	        }
    48	    }
    49	
    50	    // Signo negativo (tile 211)
    51	    if (value < 0) buf[i++] = 211;
    52	    
    53	    while (p > 0) buf[i++] = tmp[--p];
    54	    
    55	    // Limpieza con tile blanco

    60	    set_win_tiles(0, current_row++, i, 1, buf);

    64	        win_y_pos -= 8;
